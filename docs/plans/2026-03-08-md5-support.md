**Goal:** Add full MD5 (unsalted) rainbow table generation and lookup support, matching the NTLM feature set: generic kernel, fast-path kernels for 8- and 9-char plaintexts, and Markov generation.

**Architecture:** `HASH_MD5 3` constant added to `shared.h`. `CL/md5.cl` implements the MD5 hash function with prefixed macros (`MD5_F` etc.) to avoid conflicts with MD4 macros from `ntlm.cl`. `CL/rt.cl` includes `md5.cl` and dispatches to it via the existing `do_hash()` compile-time `HASH_TYPE` switch. Fast-path kernels (`md5_8_functions.cl`, `md5_9_functions.cl`) inline fixed-length padding. The existing `crackalack_markov.cl` already works for MD5 once `do_hash()` is extended - no separate Markov kernel files needed.

**Tech Stack:** C, OpenCL C, Metal (mechanically translated), libgcrypt (CPU reference via `gcry_md_hash_buffer`), RFC 1321 (MD5 spec).

### Task 2: Write CL/md5.cl - generic GPU MD5 hash function

**Files:**
- Create: `CL/md5.cl`

**Step 1: Create the file**

MD5 uses 4 auxiliary functions, 64 round constants (from RFC 1321), and a Merkle-Damgard construction. Key differences from MD4 (`ntlm.cl`): 4 rounds instead of 3, different auxiliary functions (adds `I`), different message word access order per round, add-back to state at end. Use `MD5_` prefixed macros throughout to avoid conflicts with MD4 macros already defined in `ntlm.cl`.

Create `CL/md5.cl`:
```c
/*
 * MD5 implementation for OpenCL kernels.
 * Based on RFC 1321: https://www.ietf.org/rfc/rfc1321.txt
 * Uses MD5_ prefixed macros to avoid conflicts with ntlm.cl MD4 macros.
 */

#define MD5_F(x, y, z) (((x) & (y)) | ((~x) & (z)))
#define MD5_G(x, y, z) (((x) & (z)) | ((y) & (~z)))
#define MD5_H(x, y, z) ((x) ^ (y) ^ (z))
#define MD5_I(x, y, z) ((y) ^ ((x) | (~z)))

#define MD5_STEP(f, a, b, c, d, x, t, s) \
  (a) += f((b), (c), (d)) + (x) + (uint)(t); \
  (a) = rotate((uint)(a), (uint)(s));         \
  (a) += (b);

inline void md5_compress(__private uint *state, __private uint *W) {
  uint a = state[0], b = state[1], c = state[2], d = state[3];

  /* Round 1 - F function, sequential word access */
  MD5_STEP(MD5_F, a,b,c,d, W[ 0], 0xd76aa478,  7)
  MD5_STEP(MD5_F, d,a,b,c, W[ 1], 0xe8c7b756, 12)
  MD5_STEP(MD5_F, c,d,a,b, W[ 2], 0x242070db, 17)
  MD5_STEP(MD5_F, b,c,d,a, W[ 3], 0xc1bdceee, 22)
  MD5_STEP(MD5_F, a,b,c,d, W[ 4], 0xf57c0faf,  7)

  MD5_STEP(MD5_F, c,d,a,b, W[ 6], 0xa8304613, 17)
  MD5_STEP(MD5_F, b,c,d,a, W[ 7], 0xfd469501, 22)
  MD5_STEP(MD5_F, a,b,c,d, W[ 8], 0x698098d8,  7)

  MD5_STEP(MD5_I, c,d,a,b, W[ 2], 0x2ad7d2bb, 15)
  MD5_STEP(MD5_I, b,c,d,a, W[ 9], 0xeb86d391, 21)

  state[1] += b;
  state[2] += c;
  state[3] += d;

/* Generic MD5: handles any plaintext_len up to 55 bytes (fits in one block).
 * MAX_PLAINTEXT_LEN is 16, so this is always sufficient. */
inline void md5_hash(unsigned char *plaintext, unsigned int plaintext_len,
                     unsigned char *hash) {
  uint W[16] = {0};
  uint state[4] = {0x67452301u, 0xefcdab89u, 0x98badcfeu, 0x10325476u};

  /* Pack plaintext as little-endian 32-bit words */
  for (unsigned int i = 0; i < plaintext_len; i++)
    W[i / 4] |= (uint)plaintext[i] << ((i & 3u) * 8u);

  /* Append 0x80 padding byte immediately after message */
  W[plaintext_len / 4] |= 0x80u << ((plaintext_len & 3u) * 8u);

  /* Bit-length field at word 14 (always fits: plaintext_len <= 16, bits <= 128) */
  W[14] = plaintext_len * 8u;
  /* W[15] = 0 (high 32 bits of length; always 0 here) */

  md5_compress(state, W);

  /* Write 16-byte output as little-endian bytes */
  for (int i = 0; i < 4; i++) {
    hash[i * 4 + 0] = (state[i] >>  0) & 0xffu;
    hash[i * 4 + 1] = (state[i] >>  8) & 0xffu;
    hash[i * 4 + 2] = (state[i] >> 16) & 0xffu;
    hash[i * 4 + 3] = (state[i] >> 24) & 0xffu;
  }
}
```

### Task 5: Write CL/md5_8_functions.cl - fast-path helpers for 8-char

**Files:**
- Create: `CL/md5_8_functions.cl`

**Step 1: Create the file**

This file provides three functions used by the 8-char fast-path kernels. The charset constant and `index_to_plaintext_md5_8` are identical to the ntlm8 versions (same 95-char ascii-32-95 charset). The hash function specializes MD5 for exactly 8 raw ASCII bytes (no UTF-16LE as in NTLM). The keyspace is 95^8 = 6,634,204,312,890,625.

```c
__constant char charset[] = " !\"#$%&'()*+,-./0123456789:;<=>?@ABCDEFGHIJKLMNOPQRSTUVWXYZ[\\]^_`abcdefghijklmnopqrstuvwxyz{|}~";

#include "md5.cl"


/* Identical to index_to_plaintext_ntlm8 - same base-95 decomposition. */
inline void index_to_plaintext_md5_8(unsigned long index,
                                      __constant char *charset,
                                      unsigned char *plaintext) {
  unsigned int indexHi = (unsigned int)(index / 81450625);   /* 95^4 */
  unsigned int indexLo = (unsigned int)(index - (unsigned long)81450625 * indexHi);

  unsigned short index0 = (unsigned short)(indexHi / 9025);  /* 95^2 */
  unsigned short index1 = (unsigned short)(indexHi - (unsigned int)9025 * index0);
  unsigned short index2 = (unsigned short)(indexLo / 9025);
  unsigned short index3 = (unsigned short)(indexLo - (unsigned int)9025 * index2);

  unsigned char tmp;
  tmp = (unsigned char)(index0 / 95); plaintext[0] = tmp + 32;
  plaintext[1] = (unsigned char)(index0 - (unsigned short)95 * tmp + 32);
  tmp = (unsigned char)(index1 / 95); plaintext[2] = tmp + 32;
  plaintext[3] = (unsigned char)(index1 - (unsigned short)95 * tmp + 32);
  tmp = (unsigned char)(index2 / 95); plaintext[4] = tmp + 32;
  plaintext[5] = (unsigned char)(index2 - (unsigned short)95 * tmp + 32);
  tmp = (unsigned char)(index3 / 95); plaintext[6] = tmp + 32;
  plaintext[7] = (unsigned char)(index3 - (unsigned short)95 * tmp + 32);
}


/* MD5 of exactly 8 ASCII bytes with inlined padding.
 * Block layout: W[0..1] = plaintext, W[2] = 0x80 (pad), W[3..13] = 0,
 * W[14] = 64 (bit length), W[15] = 0.
 * Returns first 8 bytes of hash as little-endian uint64. */
inline unsigned long hash_md5_8(unsigned char *plaintext) {
  uint W[16] = {0};
  uint state[4] = {0x67452301u, 0xefcdab89u, 0x98badcfeu, 0x10325476u};

  W[0] = (uint)plaintext[0] | ((uint)plaintext[1] << 8)
       | ((uint)plaintext[2] << 16) | ((uint)plaintext[3] << 24);
  W[1] = (uint)plaintext[4] | ((uint)plaintext[5] << 8)
       | ((uint)plaintext[6] << 16) | ((uint)plaintext[7] << 24);
  W[2] = 0x00000080u;  /* 0x80 pad byte at position 8 */
  W[14] = 64u;         /* 8 bytes * 8 bits */

  md5_compress(state, W);

  return (unsigned long)state[0] | ((unsigned long)state[1] << 32);
}


inline unsigned long hash_to_index_md5_8(unsigned long hash, unsigned int pos) {
  return (hash + pos) % 6634204312890625UL;  /* 95^8 */
}


inline unsigned long hash_char_to_index_md5_8(__global unsigned char *hash_value,
                                               unsigned int pos) {
  unsigned long ret = hash_value[7]; ret <<= 8;
  ret |= hash_value[6]; ret <<= 8;
  ret |= hash_value[5]; ret <<= 8;
  ret |= hash_value[4]; ret <<= 8;
  ret |= hash_value[3]; ret <<= 8;
  ret |= hash_value[2]; ret <<= 8;
  ret |= hash_value[1]; ret <<= 8;
  ret |= hash_value[0];
  return (ret + pos) % 6634204312890625UL;
}
```

**Step 1: Create crackalack_md5_8.cl**

The chain length for the ascii-32-95 8-char MD5 fast path is **422000** (same keyspace as NTLM8, same recommended chain length). The loop runs `chain_len - 1 = 421999` iterations.

```c
#include "md5_8_functions.cl"

__kernel void crackalack_md5_8(
    __global unsigned int *unused1, __global char *unused2,
    __global unsigned int *unused3, __global unsigned int *unused4,
    __global unsigned int *unused5, __global unsigned int *unused6,
    __global unsigned long *g_indices, __global unsigned int *unused7,
    __global unsigned long *unused8, __global unsigned long *unused9,
    __global unsigned int *unused10, __global char *unused11,
    __global unsigned int *unused12) {

  unsigned long index = g_indices[get_global_id(0)];
  unsigned char plaintext[8];

  for (unsigned int pos = 0; pos < 421999; pos++) {
    index_to_plaintext_md5_8(index, charset, plaintext);
    index = hash_to_index_md5_8(hash_md5_8(plaintext), pos);
  }

  g_indices[get_global_id(0)] = index;
}
```

```c
__kernel void precompute_md5_8(
    __global unsigned int *unused1, __global unsigned char *g_hash,
    __global unsigned int *unused2, __global char *unused3,
    __global unsigned int *unused4, __global unsigned int *unused5,

    index_to_plaintext_md5_8(index, charset, plaintext);
    index = hash_to_index_md5_8(hash_md5_8(plaintext), i);
```

```c
    __global unsigned long *g_plaintext_indices) {

  int index_pos = (*g_num_start_indices - *g_device_num)
      - ((get_global_id(0) + *g_exec_block_scaler) * *g_total_devices) - 1;
  if (index_pos < 0)
    return;

  unsigned char plaintext[8];
  unsigned long index = g_start_indices[index_pos], previous_index = 0;
  unsigned long hash_base_index = g_hash_base_indices[index_pos] % 6634204312890625UL;
  unsigned int endpoint = g_start_index_positions[index_pos];

  for (unsigned int pos = 0; pos <= endpoint; pos++) {
    index_to_plaintext_md5_8(index, charset, plaintext);
    previous_index = index;
    index = hash_to_index_md5_8(hash_md5_8(plaintext), pos);

    if ((index == (hash_base_index + pos)) ||
        (index == (hash_base_index + pos - 6634204312890625UL))) {
      g_plaintext_indices[index_pos] = previous_index;
      return;
    }
  }
}
```

**Files:** Parallel set to Tasks 5-8, with these key differences:

| Parameter | md5_8 | md5_9 |
|-----------|-------|-------|
| Plaintext length | 8 | 9 |

| Chain loop iterations | 421,999 | 802,999 |

| Kernel/function names | `*_md5_8` | `*_md5_9` |

**W[2] for 9-char:** byte 8 is the last plaintext byte, byte 9 is the 0x80 pad:
```c
W[2] = (uint)plaintext[8] | (0x80u << 8);  /* p[8] then 0x80 */
```

**Index decomposition for 9-char:** 95^9 doesn't factor as cleanly as 95^8. Use a direct approach:
```c
inline void index_to_plaintext_md5_9(unsigned long index,
                                      __constant char *charset,
                                      unsigned char *plaintext) {
  for (int i = 8; i >= 0; i--) {
    plaintext[i] = (unsigned char)(index % 95) + 32;
    index /= 95;
  }
}
```

**Key Metal translation rules (see existing Metal files for reference):**

1. `__kernel` -> `kernel`

5. `unsigned long` -> `ulong` (CRITICAL: `unsigned long` is 32-bit in Metal, `ulong` is 64-bit)
